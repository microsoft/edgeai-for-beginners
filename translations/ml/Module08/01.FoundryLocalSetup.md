def create_client():

    # Azure OpenAI เดเตเตบเดซเดฟเดเดฑเตเดทเตป เดชเดฐเดฟเดถเตเดงเดฟเดเตเดเตเด

    alias = os.environ.get("MODEL", "phi-4-mini")
