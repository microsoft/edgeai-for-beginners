BitNET เดฎเตเดกเดฒเตเดเตพเดเตเดเต Olive เดเดเดจเดฏเดฟเตฝ เดชเตเดฐเดคเตเดฏเตเด 1-เดฌเดฟเดฑเตเดฑเต เดเตเดตเดพเดฃเตเดเตเดธเตเดทเตป เดชเตเดฐเดตเตเดคเตเดคเดฟเดเตพ เดเดตเดถเตเดฏเดฎเดพเดฃเต.

**เดชเตเดฐเดตเตเดคเตเดคเดฟ เดเดเตเดเตเดฎเตเดทเตป**: เดฎเตเดเตเดเดชเตเดชเตเดเตเดคเตเดคเตฝ เดตเดเดญเตเดฆเดเตเดเดณเดฟเตฝ เดเตเดฃเดจเดฟเดฒเดตเดพเดฐ เดฎเดพเดจเดฆเดฃเตเดกเด เดธเดเดฐเดเตเดทเดฟเดเตเดเตเดจเตเดจเดคเดฟเดจเตเดณเตเดณ เดธเตเดตเดฏเด เดฌเดเตเดเตเดฎเดพเตผเดเตเดเดฟเดเดเต. PyTorch, ONNX เดชเตเดฒเตเดณเตเดณ เดชเตเดฐเดถเดธเตเดค ML เดเดเดจเดเดณเตเดฎเดพเดฏเดฟ เดธเดเดฏเตเดเดจเด, เดเตเดฒเตเดกเต, เดเดกเตเดเต เดตเดฟเดจเตเดฏเดพเดธ เดฎเตเดเตเดเดชเตเดชเตเดเตเดคเตเดคเตฝ เดเดดเดฟเดตเตเดเตพ.

**เดชเตเดฐเดพเดฏเตเดเดฟเด เดจเดเดชเตเดชเดพเดเตเดเตฝ เดเดฆเดพเดนเดฐเดฃเด**:

```python
# SLM-เดจเตเดณเตเดณ Microsoft Olive เดเดชเตเดฑเตเดฑเดฟเดฎเตเดธเตเดทเตป เดตเตผเดเตเดเตโเดซเตเดฒเต
from olive.model import PyTorchModel, ONNXModel
from olive.workflows import run_workflow
from transformers import AutoModelForCausalLM, AutoTokenizer
import torch

# เดตเตผเดเตเดเตโเดซเตเดฒเต เดเตเตบเดซเดฟเดเดฑเตเดทเตป เดจเดฟเตผเดตเดเดฟเดเตเดเตเด
def create_olive_config(model_id="microsoft/phi-4-mini-instruct"):
    # เดฎเตเดกเตฝ เดฒเตเดกเต เดเตเดฏเตเดคเต เดธเดพเดฎเตเดชเดฟเตพ เดเตปเดชเตเดเตเดเตเดเตพ เดธเตเดทเตเดเดฟเดเตเดเตเด
    tokenizer = AutoTokenizer.from_pretrained(model_id)
    model = AutoModelForCausalLM.from_pretrained(model_id, torch_dtype=torch.float16)
    
    # เดเตเดฐเตเดธเดฟเดเดเดฟเดจเดพเดฏเดฟ เดธเดพเดฎเตเดชเดฟเตพ เดเตปเดชเตเดเตเดเตเดเตพ เดธเตเดทเตเดเดฟเดเตเดเตเด
    sample_text = "Explain the concept of edge computing"
    inputs = tokenizer(sample_text, return_tensors="pt")
    
    # เดเดฆเตเดฏเด ONNX-เดฒเตเดเตเดเต เดเดเตเดธเตเดชเตเตผเดเตเดเต เดเตเดฏเตเดฏเตเด
    model_path = f"{model_id.split('/')[-1]}.onnx"
    torch.onnx.export(
        model,
        (inputs["input_ids"],),
        model_path,
        input_names=["input_ids"],
        output_names=["logits"],
        dynamic_axes={
            "input_ids": {0: "batch", 1: "sequence"},
            "logits": {0: "batch", 1: "sequence"}
        },
        opset_version=15
    )
    
    # Olive เดเดชเตเดฑเตเดฑเดฟเดฎเตเดธเตเดทเตป เดเตเตบเดซเดฟเดเต เดธเตเดทเตเดเดฟเดเตเดเตเด
    config = {
        "input_model": ONNXModel(model_path),
        "systems": {
            "local_system": {
                "type": "LocalSystem"
            }
        },
        "passes": {
            # เดเตเดฐเดพเดซเต เดเดชเตเดฑเตเดฑเดฟเดฎเตเดธเตเดทเตป เดชเดพเดธเต
            "graph_optimization": {
                "type": "OrtTransformersOptimization",
                "config": {
                    "optimization_options": {
                        "enable_gelu": True,
                        "enable_layer_norm": True,
                        "enable_attention": True,
                        "use_multi_head_attention": True
                    }
                }
            },
            # INT8-เดจเตเดณเตเดณ เดเตเดตเดพเดฃเตเดเตเดธเตเดทเตป เดชเดพเดธเต
            "quantization": {
                "type": "OrtQuantization",
                "config": {
                    "quant_mode": "static",
                    "activation_type": "int8",
                    "weight_type": "int8",
                    "op_types_to_quantize": ["MatMul", "Add", "Conv"]
                },
                "disable_search": True
            }
        },
        "engine": {
            "log_severity_level": 0,
            "cache_dir": "./cache"
        }
    }
    
    return config
```
