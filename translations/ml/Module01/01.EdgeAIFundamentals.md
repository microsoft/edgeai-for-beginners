```python

import torch

# เดฎเตเตปเดเตเดเตเดเดฟ เดชเดฐเดฟเดถเตเดฒเดฟเดเตเด เดฎเตเดกเตฝ เดฒเตเดกเต เดเตเดฏเตเดฏเตเด
model = torch.load('large_model.pth')

# เดฎเตเดกเตฝ INT8 เดเดฏเดฟ เดเตเดตเดพเดฃเตเดเตเดธเต เดเตเดฏเตเดฏเตเด
quantized_model = torch.quantization.quantize_dynamic(
    model,  # เดเตเดตเดพเดฃเตเดเตเดธเต เดเตเดฏเตเดฏเดพเดจเตเดณเตเดณ เดฎเตเดกเตฝ
    {torch.nn.Linear, torch.nn.Conv2d},  # เดเตเดตเดพเดฃเตเดเตเดธเต เดเตเดฏเตเดฏเดพเดจเตเดณเตเดณ เดฒเตเดฏเดฑเตเดเตพ
    dtype=torch.qint8  # เดเตเดตเดพเดฃเตเดเตเดธเตเดทเตป เดกเดพเดฑเตเดฑเดพ เดเตเดชเตเดชเต

# เดเตเดตเดพเดฃเตเดเตเดธเต เดเตเดฏเตเดค เดฎเตเดกเตฝ เดธเตเดตเต เดเตเดฏเตเดฏเตเด

# เดฎเตเดฎเตเดฎเดฑเดฟ เดเดชเดฏเตเดเด เดคเดพเดฐเดคเดฎเตเดฏเด
```

เดตเดฟเดตเดฟเดง เดเตเดตเดพเดฃเตเดเตเดธเตเดทเตป เดธเดพเดเตเดเตเดคเดฟเดเดตเดฟเดฆเตเดฏเดเตพ เดเตพเดชเตเดชเตเดเตเดจเตเดจเต:

- **เดธเตเดฑเตเดฑเดพเดฑเตเดฑเดฟเดเต เดเตเดตเดพเดฃเตเดเตเดธเตเดทเตป**: เดญเดพเดฐเดเตเดเตพเดเตเดเตเด เดเดเตเดเดฟเดตเตเดทเดจเตเดเตพเดเตเดเตเด เดเดฒเตเดฒเดพ เดเตเดตเดพเดฃเตเดเตเดธเตเดทเตป เดชเดพเดฐเดพเดฎเตเดฑเตเดฑเดฑเตเดเดณเตเด เดฎเตเตปเดเตเดเตเดเดฟ เดเดฃเดเตเดเดพเดเตเดเตเดจเตเดจเต

```python
pruned_model.fit(x_train, y_train, epochs=10)
```

### เดเตเดฑเดฟเดฏ เดญเดพเดทเดพ เดฎเตเดกเดฒเตเดเตพ (SLMs)
