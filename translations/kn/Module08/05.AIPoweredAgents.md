```python
# agent.py
from foundry_local import FoundryLocalManager
from openai import OpenAI
import json
from tools import TOOLS, get_weather

# ‡≤´‡≥å‡≤Ç‡≤°‡≥ç‡≤∞‡≤ø ‡≤≤‡≥ã‡≤ï‡≤≤‡≥ç ‡≤Æ‡≥ç‡≤Ø‡≤æ‡≤®‡≥á‡≤ú‡≤∞‡≥ç ‡≤Ö‡≤®‡≥ç‡≤®‡≥Å ‡≤™‡≥ç‡≤∞‡≤æ‡≤∞‡≤Ç‡≤≠‡≤ø‡≤∏‡≤ø
alias = "phi-4-mini"
manager = FoundryLocalManager(alias)

# ‡≤´‡≥å‡≤Ç‡≤°‡≥ç‡≤∞‡≤ø ‡≤≤‡≥ã‡≤ï‡≤≤‡≥ç ‡≤é‡≤Ç‡≤°‡≥ç‚Äå‡≤™‡≤æ‡≤Ø‡≤ø‡≤Ç‡≤ü‡≥ç ‡≤¨‡≤≥‡≤∏‡≤ø OpenAI ‡≤ï‡≥ç‡≤≤‡≥à‡≤Ç‡≤ü‡≥ç ‡≤∞‡≤ö‡≤ø‡≤∏‡≤ø
client = OpenAI(
    base_url=manager.endpoint,
    api_key=manager.api_key
)

SYSTEM_PROMPT = "You are a helpful assistant. Use tools when needed."

def process_function_call(messages: List[Dict], tools: List[Dict]) -> str:
    """Process function calling with modern OpenAI API."""
    try:
        response = client.chat.completions.create(
            model=manager.get_model_info(alias).id,
            messages=messages,
            tools=tools,
            tool_choice="auto"
        )
        
        message = response.choices[0].message
        
        if message.tool_calls:
            # ‡≤´‡≤Ç‡≤ï‡≥ç‡≤∑‡≤®‡≥ç ‡≤ï‡≤∞‡≥Ü‡≤ó‡≤≥‡≤®‡≥ç‡≤®‡≥Å ‡≤®‡≤ø‡≤∞‡≥ç‡≤µ‡≤π‡≤ø‡≤∏‡≤ø
            messages.append(message)
            
            for tool_call in message.tool_calls:
                if tool_call.function.name == "get_weather":
                    args = json.loads(tool_call.function.arguments)
                    result = get_weather(args["city"])
                    
                    # ‡≤∏‡≤Ç‡≤¶‡≥á‡≤∂‡≤ó‡≤≥‡≤ø‡≤ó‡≥Ü ‡≤´‡≤Ç‡≤ï‡≥ç‡≤∑‡≤®‡≥ç ‡≤´‡≤≤‡≤ø‡≤§‡≤æ‡≤Ç‡≤∂‡≤µ‡≤®‡≥ç‡≤®‡≥Å ‡≤∏‡≥á‡≤∞‡≤ø‡≤∏‡≤ø
                    messages.append({
                        "role": "tool",
                        "tool_call_id": tool_call.id,
                        "content": result
                    })
            
            # ‡≤Ö‡≤Ç‡≤§‡≤ø‡≤Æ ‡≤™‡≥ç‡≤∞‡≤§‡≤ø‡≤ï‡≥ç‡≤∞‡≤ø‡≤Ø‡≥Ü‡≤Ø‡≤®‡≥ç‡≤®‡≥Å ‡≤™‡≤°‡≥Ü‡≤Ø‡≤ø‡≤∞‡≤ø
            final_response = client.chat.completions.create(
                model=manager.get_model_info(alias).id,
                messages=messages
            )
            return final_response.choices[0].message.content
        else:
            return message.content
            
    except Exception as e:
        return f"Error: {str(e)}"

# ‡≤â‡≤¶‡≤æ‡≤π‡≤∞‡≤£‡≥Ü‡≤Ø ‡≤¨‡≤≥‡≤ï‡≥Ü
messages = [
    {"role": "system", "content": SYSTEM_PROMPT},
    {"role": "user", "content": "What's the weather in Paris?"}
]

result = process_function_call(messages, TOOLS)
print(result)
```
