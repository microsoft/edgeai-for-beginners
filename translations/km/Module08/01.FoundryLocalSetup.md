# សម័យ 1: ការចាប់ផ្តើមជាមួយ Foundry Local

## សេចក្តីសង្ខេប

Microsoft Foundry Local នាំមកនូវសមត្ថភាព Azure AI Foundry ដល់បរិយាកាសអភិវឌ្ឍន៍ Windows 11 របស់អ្នក ដើម្បីអភិវឌ្ឍ AI ដែលរក្សាភាពឯកជន និងLatency ទាប ជាមួយឧបករណ៍ថ្នាក់សហគ្រាស។ សម័យនេះគ្របដណ្តប់ពីការដំឡើង ការកំណត់រចនាសម្ព័ន្ធ រហូតដល់ការដាក់បញ្ចូលមូឌែលប្រជាប្រិយរួមមាន phi, qwen, deepseek និង GPT-OSS-20B។

## គោលបំណងការសិក្សា

ចប់សម័យនេះ អ្នកនឹងជា:
- ដំឡើង និងកំណត់រចនាសម្ព័ន្ធ Foundry Local លើ Windows 11
- ចេះជំនាញបញ្ជា CLI និងជម្រើសកំណត់រចនាសម្ព័ន្ធ
- យល់ពីយុទ្ធសាស្ត្រការផ្ទុកមូឌែល (caching) ដើម្បីបង្កើនប្រសិទ្ធភាព
- រត់បានយ៉ាងជោគជ័យនូវមូឌែល phi, qwen, deepseek និង GPT-OSS-20B
- បង្កើតកម្មវិធី AI លើកដំបូងរបស់អ្នកដោយប្រើ Foundry Local

## ផ្នែក 5: បង្កើតកម្មវិធីដំបូងរបស់អ្នក

### កម្មវិធីជជែកទំនើប (OpenAI SDK + Foundry Local)

បង្កើតកម្មវិធីជជែកសម្រាប់ផលិតកម្ម ដោយប្រើ OpenAI SDK ជាមួយការរួមបញ្ចូល Foundry Local អនុវត្តតាមគំរូពី Sample 01។

```python
# chat_quickstart.py (លំនាំគំរូ 01)
import os
import sys
from openai import OpenAI

try:
    from foundry_local import FoundryLocalManager
    FOUNDRY_SDK_AVAILABLE = True
except ImportError:
    FOUNDRY_SDK_AVAILABLE = False
    print("⚠️ Install foundry-local-sdk: pip install foundry-local-sdk")

def create_client():
    """Create OpenAI client with Foundry Local or Azure OpenAI."""
    # ពិនិត្យការកំណត់រចនាសម្ព័ន្ធ Azure OpenAI
    azure_endpoint = os.environ.get("AZURE_OPENAI_ENDPOINT")
    azure_api_key = os.environ.get("AZURE_OPENAI_API_KEY")
    
    if azure_endpoint and azure_api_key:
        # ផ្លូវ Azure OpenAI
        model = os.environ.get("MODEL", "your-deployment-name")
        client = OpenAI(
            base_url=f"{azure_endpoint}/openai",
            api_key=azure_api_key,
            default_query={"api-version": "2024-08-01-preview"},
        )
        print(f"🌐 Using Azure OpenAI with model: {model}")
        return client, model
    
    # ផ្លូវ Foundry Local ជាមួយការគ្រប់គ្រង SDK
    alias = os.environ.get("MODEL", "phi-4-mini")
    if FOUNDRY_SDK_AVAILABLE:
        try:
            # ប្រើ FoundryLocalManager សម្រាប់ការគ្រប់គ្រងសេវាកម្មដែលត្រឹមត្រូវ
            manager = FoundryLocalManager(alias)
            model_info = manager.get_model_info(alias)
            
            client = OpenAI(
                base_url=manager.endpoint,
                api_key=manager.api_key
            )
            model = model_info.id
            print(f"🏠 Using Foundry Local SDK with model: {model}")
            return client, model
        except Exception as e:
            print(f"⚠️ Foundry SDK failed ({e}), using manual configuration")
    
    # ត្រឡប់ទៅកាន់ការកំណត់ដោយដៃ
    base_url = os.environ.get("BASE_URL", "http://localhost:8000")
    api_key = os.environ.get("API_KEY", "")
    model = alias
    
    client = OpenAI(
        base_url=f"{base_url}/v1",
        api_key=api_key
    )
    print(f"🔧 Manual configuration with model: {model}")
    return client, model

def main():
    """Main chat function."""
    client, model = create_client()
    
    print("Foundry Local Chat Interface (type 'quit' to exit)\n")
    conversation_history = []
    
    while True:
        user_input = input("You: ")
        if user_input.lower() == 'quit':
            break
        
        try:
            # បន្ថែមសាររបស់អ្នកប្រើទៅក្នុងប្រវត្តិ
            conversation_history.append({"role": "user", "content": user_input})
            
            # បង្កើតការបញ្ចប់សម្រាប់ការជជែក
            response = client.chat.completions.create(
                model=model,
                messages=conversation_history,
                max_tokens=500,
                temperature=0.7
            )
            
            assistant_message = response.choices[0].message.content
            conversation_history.append({"role": "assistant", "content": assistant_message})
            
            print(f"Assistant: {assistant_message}\n")
            
        except Exception as e:
            print(f"Error: {e}\n")

if __name__ == "__main__":
    main()
```

### អនុវត្តិល្អបំផុត

- **Use OpenAI SDK**: ប្រើ OpenAI SDK ជាជម្រើសល្អជាងការបញ្ចូនសំណើ HTTP ដោយផ្ទាល់ ដើម្បីងាយស្រួលក្នុងការថែរក្សា
- **FoundryLocalManager**: ប្រើ SDK ផ្លូវការសម្រាប់ការគ្រប់គ្រងសេវាកម្ម បើមាន
- **Error Handling**: អនុវត្តយុទ្ធសាស្ត្រ fallback សមរម្យសម្រាប់កម្មវិធីក្នុងផលិតកម្ម
- **Upgrade Regularly**: រក្សា Foundry Local ឲ្យទាន់សម័យ ដើម្បីចូលដំណើរការមូឌែលថ្មី និងការស្ដារជួសជុល
- **Start Small**: ចាប់ផ្តើមពីមូឌែលតូចៗ (Phi mini, Qwen 7B) ហើយពង្រីកក្រោយ
- **Monitor Resources**: តាមដាន CPU/GPU/ម៉េម៉ូរី ខណៈកំពុងលៃតម្រូវ prompts និងការកំណត់

### លំហាត់ 2: សាកល្បងការរួមបញ្ចូល OpenAI SDK

```python
# sdk_integration_test.py (ស្របទៅនឹងលំនាំ Sample 01)
import os
from openai import OpenAI
from foundry_local import FoundryLocalManager

def test_model_integration(model_alias):
    """Test OpenAI SDK integration with different models."""
    try:
        # ប្រើ FoundryLocalManager សម្រាប់ការរៀបចំដែលត្រឹមត្រូវ
        manager = FoundryLocalManager(model_alias)
        model_info = manager.get_model_info(model_alias)
        
        client = OpenAI(
            base_url=manager.endpoint,
            api_key=manager.api_key
        )
        
        # សាកល្បងការបញ្ចប់មូលដ្ឋាន
        response = client.chat.completions.create(
            model=model_info.id,
            messages=[{"role": "user", "content": "Say hello and state your model name."}],
            max_tokens=50
        )
        
        print(f"✅ {model_alias}: {response.choices[0].message.content}")
        return True
    except Exception as e:
        print(f"❌ {model_alias}: {e}")
        return False

# សាកល្បងម៉ូដែលច្រើន
models_to_test = ["phi-4-mini", "qwen2.5-7b"]
for model in models_to_test:
    test_model_integration(model)
```

### លំហាត់ 3: ពិនិត្យសុខភាពសេវាកម្មទូលំទូលាយ

```python
# health_check.py
from openai import OpenAI
from foundry_local import FoundryLocalManager

def comprehensive_health_check():
    """Perform comprehensive health check of Foundry Local service."""
    try:
        # ចាប់ផ្តើមជាមួយម៉ូឌែលទូទៅ
        manager = FoundryLocalManager("phi-4-mini")
        client = OpenAI(
            base_url=manager.endpoint,
            api_key=manager.api_key
        )
        
        # ១. ពិនិត្យការតភ្ជាប់សេវាកម្ម
        models_response = client.models.list()
        available_models = [model.id for model in models_response.data]
        print(f"✅ Service healthy - {len(available_models)} models available")
        
        # ២. សាកល្បងម៉ូឌែលដែលអាចប្រើបាននីមួយៗ
        for model_id in available_models:
            try:
                response = client.chat.completions.create(
                    model=model_id,
                    messages=[{"role": "user", "content": "Test"}],
                    max_tokens=10
                )
                print(f"✅ {model_id}: Working")
            except Exception as e:
                print(f"❌ {model_id}: {e}")
        
        return True
    except Exception as e:
        print(f"❌ Service check failed: {e}")
        return False

comprehensive_health_check()
```
