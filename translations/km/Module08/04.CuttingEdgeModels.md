### Key Technologies

- `FoundryLocalManager(alias)`: Automatic service management

### Hybrid Strategy Patterns

**Local-First with Fallback:**
```python
async def hybrid_completion(prompt: str, complexity_threshold: int = 100):
    if len(prompt.split()) < complexity_threshold:
        return await local_completion(prompt)  # бһӣбһҝбһ“, бһҜбһҖбһҮбһ“
    else:
        return await cloud_completion(prompt)   # бһҖбһ¶бһҡбһӮбһ·бһҸбһҹбҹ’бһҳбһ»бһӮбһҹбҹ’бһҳбһ¶бһү
```

**Task-Based Routing:**
```python
async def smart_routing(prompt: str, task_type: str):
    routing_rules = {
        "code_generation": "local",     # бһ‘бһ“бҹӢбһ—бҹ’бһӣбһ“бҹӢбһ…бҹҶбһ–бҹ„бҹҮбһ—бһ¶бһ–бһҜбһҖбһҮбһ“
        "creative_writing": "cloud",    # бһ‘бһ‘бһҪбһӣбһ”бһ¶бһ“бһўбһҸбҹ’бһҗбһ”бҹ’бһҡбһҷбҹ„бһҮбһ“бҹҚбһ–бһёбһҳбҹүбһјбһҢбҹӮбһӣбһҠбҹӮбһӣбһ’бҹҶбһҮбһ¶бһ„
        "data_analysis": "local",       # бһҸбҹ’бһҡбһјбһңбһҖбһ¶бһҡбһҖбһ¶бһҡбһҖбҹӮбһ”бҹ’бһҡбҹӮбһҷбҹүбһ¶бһ„бһҡбһ бҹҗбһҹ
        "research": "cloud"             # бһҸбҹ’бһҡбһјбһңбһҖбһ¶бһҡбһ…бҹҶбһҺбҹҒбҹҮбһҠбһ№бһ„бһ‘бһјбһӣбҹҶбһ‘бһјбһӣбһ¶бһҷ
    }
    
    if routing_rules.get(task_type) == "local":
        return await foundry_completion(prompt)
    else:
        return await azure_completion(prompt)
```

```python
    alias = os.environ.get("MODEL", "phi-4-mini")
    
    try:
        # бһ”бҹ’бһҡбһҫ FoundryLocalManager бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҖбһ¶бһҡбһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһҹбҹҒбһңбһ¶бһҖбһҳбҹ’бһҳбһҷбҹүбһ¶бһ„бһҸбҹ’бһҡбһ№бһҳбһҸбҹ’бһҡбһјбһң
        manager = FoundryLocalManager(alias)
        model_info = manager.get_model_info(alias)
        
        client = OpenAI(
            base_url=manager.endpoint,
            api_key=manager.api_key or "not-required"
        )
        model_name = model_info.id if model_info else alias
        return True
    except Exception as e:
        # бһңбһ·бһӣбһҸбҹ’бһҡбһЎбһ”бҹӢбһ‘бҹ…бһҖбһ¶бһҡбһҖбҹҶбһҺбһҸбҹӢбһҠбҹ„бһҷбһҠбҹғ
        base_url = os.environ.get("BASE_URL", "http://localhost:51211")
        client = OpenAI(base_url=f"{base_url}/v1", api_key="not-required")
        model_name = alias
        return True
```

**Streaming Chat Handler:**
```python
@cl.on_message
async def main(message: cl.Message):
    # бһ”бһ„бҹ’бһҖбһҫбһҸбһ…бһҳбҹ’бһӣбһҫбһҷбһҠбҹӮбһӣбһ•бҹ’бһҹбһ¶бһҷбһ”бһ“бҹ’бһҸ
    msg = cl.Message(content="")
    await msg.send()
    
    stream = client.chat.completions.create(
        model=model_name,
        messages=[
            {"role": "system", "content": "You are a helpful AI assistant."},
            {"role": "user", "content": message.content}
        ],
        stream=True
    )
    
    # бһ•бҹ’бһҹбһ¶бһҷбһҸбһјбһҖбһ·бһ“бһҖбҹ’бһ“бһ»бһ„бһ–бҹҒбһӣбһ–бһ·бһҸ
    for chunk in stream:
        if chunk.choices[0].delta.content:
            await msg.stream_token(chunk.choices[0].delta.content)
    
    await msg.update()
```

```javascript
const outputEl = document.getElementById('output');

function log(msg) {
    outputEl.textContent += `${msg}\n`;
    console.log(msg);
}

(async () => {
    try {
        if (!('gpu' in navigator)) {
            statusEl.textContent = 'вқҢ WebGPU not available';
            return;
        }
        
        statusEl.textContent = 'рҹ”Қ WebGPU detected. Loading model...';
        
        // бһ”бҹ’бһҡбһҫбһҳбҹүбһјбһҢбҹӮбһӣ ONNX бһҸбһјбһ…бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһ”бһ„бҹ’бһ бһ¶бһү
        const modelUrl = 'https://huggingface.co/onnx/models/resolve/main/vision/classification/mnist-12/mnist-12.onnx';
        
        const session = await ort.InferenceSession.create(modelUrl, {
            executionProviders: ['webgpu']
        });
        
        log('вң… ONNX Runtime session created with WebGPU');
        log(`рҹ“Ҡ Input names: ${session.inputNames.join(', ')}`);
        log(`рҹ“Ҡ Output names: ${session.outputNames.join(', ')}`);

        const input = new ort.Tensor('float32', inputData, [1, 1, 28, 28]);

    }
```

Session 4 бһӮбһәбһӮбҹ’бһҡбһ”бһҠбһҺбҹ’бһҸбһ”бҹӢбһҖбһ¶бһҡбһҹбһ¶бһ„бһҹбһ„бҹӢбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһё Chainlit бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢ AI бһҹбһ“бҹ’бһ‘бһ“бһ¶бһҠбҹӮбһӣбһҹбһ¶бһҖбһҹбһҳбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһ•бһӣбһ·бһҸбһҖбһҳбҹ’бһҳбҹ” бһўбҹ’бһ“бһҖбһ”бһ¶бһ“бһҡбҹҖбһ“бһўбҹҶбһ–бһё:

- вң… **Chainlit Framework**: UI бһҹбһҳбҹҗбһҷбһ‘бҹҶбһ“бһҫбһ” бһ“бһ·бһ„бһҖбһ¶бһҡбһӮбһ¶бҹҶбһ‘бҹ’бһҡбһ…бһҳбҹ’бһӣбһҫбһҷбһ”бһ“бҹ’бһҸбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһҮбһҮбҹӮбһҖ
- вң… **Foundry Local Integration**: бһҖбһ¶бһҡбһ”бҹ’бһҡбһҫбһ”бҹ’бһҡбһ¶бһҹбҹӢ SDK бһ“бһ·бһ„бһӣбҹҶбһ“бһ¶бҹҶбһҖбҹҶбһҺбһҸбҹӢбһҡбһ…бһ“бһ¶бһҹбһҳбҹ’бһ–бҹҗбһ“бҹ’бһ’  
- вң… **WebGPU Inference**: AI бһ“бҹ…бһҖбҹ’бһ“бһ»бһ„бһҖбһҳбҹ’бһҳбһңбһ·бһ”бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҜбһҖбһҮбһ“бһ—бһ¶бһ–бһўбһҸбһ·бһ”бһҡбһ·бһҳбһ¶
- вң… **Open WebUI Setup**: бһҖбһ¶бһҡбһҠбһ¶бһҖбҹӢбһҠбҹҶбһҺбһҫбһҡбһҖбһ¶бһҡбһ•бҹ’бһ‘бһ¶бҹҶбһ„бһҮбһҮбҹӮбһҖбһҮбһ¶бһ”бҹӢбһҮбҹҶбһ“бһ¶бһү
- вң… **Production Patterns**: бһҖбһ¶бһҡбһҠбҹ„бҹҮбһҹбҹ’бһҡбһ¶бһҷбһҖбҹҶбһ бһ»бһҹ, бһҖбһ¶бһҡбһҸбҹ’бһҡбһҪбһҸбһ–бһ·бһ“бһ·бһҸбҹ’бһҷ бһ“бһ·бһ„бһҖбһ¶бһҡбһҖбҹҶбһҺбһҫбһ“
